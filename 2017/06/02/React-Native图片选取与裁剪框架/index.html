安装

iOS

  • 在 Xcode 的 Info.plist 中添加：

    1. NSPhotoLibraryUsageDescription //相册
    2. NSCameraUsageDescription //摄像头
    3. NSMicrophoneUsageDescription //麦克风

  • cocoapods 用户

    1. Add platform :ios, '8.0' to Podfile (!important)
    2. Add pod 'RSKImageCropper' and pod 'QBImagePickerController' to Podfile

  • 非 cocoapods 用户

    1. Under Deployment Info set Deployment Target to 8.0
    2. Under Embedded Binaries click + and add RSKImageCropper.framework and QBImagePicker.framework

Android

  • Make sure you are using Gradle 2.2.x (project build.gradle)
  • Add useSupportLibrary (app build.gradle)
  • [Optional] If you want to use camera picker in your project, add following to AndroidManifest.xml
  • <uses-permission android:name="android.permission.CAMERA"/>

代码示例

import React, {Component} from 'react';
import {
  TouchableHighlight,
  ListView,
  View,
  StyleSheet,
  Image,
  Text,
} from 'react-native';
import {StackNavigator} from 'react-navigation';
import ImagePicker from 'react-native-image-crop-picker';

export default class JLPersonCenter extends Component {

  constructor(){
    super();
    const ds=new ListView.DataSource({rowHasChanged:(r1, r2) => r1 !== r2});
    this.state={
      dataSource: ds.cloneWithRows(['修改密码','意见反馈','关于']),
      headerPath: require('../Source/HomeHeader.jpg'),
    };
  }

  _renderRow(rowData: string, sectionID: number, rowID: number, highlightRow: (sectionID: number, rowID: number) => void){
    return(
      <View style={styles.rowStyle}>
        <Text style={styles.rowText}>{rowData}</Text>
      </View>
    );
  }

  _renderSeparator(){
    return(
      <View
        style={{height: 0.5, marginLeft: 5,marginRight: 5, backgroundColor: 'gray'}}>
      </View>
    );
  }

  _renderHeader(){
    return(
      <View style={styles.headerStyle}>
        <TouchableHighlight style={{borderRadius: 40}} underlayColor='#C4C3C5' onPress={() => {
          this._exchangeHeaderImage();
        }}>
        <Image
          source={this.state.headerPath}
          style={{width: 80, height: 80,borderRadius: 40, borderWidth: 2, borderColor: '#EDECF0'}}
        />
        </TouchableHighlight>
        <Text style={{fontSize: 15, textAlign: 'center', color: '#EDECF0',paddingTop: 5}}>欧阳雨露</Text>
      </View>
    );
  }

  // 更换头像
  _exchangeHeaderImage(){
    ImagePicker.openPicker({
      cropping: true,
      cropperCircleOverlay: true,
      width: 160,
      height: 160,
    }).then(image => {
      this.setState({headerPath:{uri: image.path}});
    });
  }

  render(){
    return(
      <ListView
        removeClippedSubviews={false}
        dataSource={this.state.dataSource}
        renderRow={this._renderRow.bind(this)}
        renderSeparator={this._renderSeparator}
        renderHeader={this._renderHeader.bind(this)}
      />
    );
  }
}

const styles = StyleSheet.create({
  icon: {
    width: 26,
    height: 26,
  },
  headerStyle: {
    flex: 1,
    height: 160,
    backgroundColor:'#5EBA6A',
    justifyContent: 'center',
    alignItems: 'center',
  },
  rowStyle: {
    flex: 1,
    flexDirection: 'column',
    padding: 5,
    height: 45,
    backgroundColor: '#F6F6F6',
  },
  rowText: {
    flex: 1,
    lineHeight: 35,
  },
});
